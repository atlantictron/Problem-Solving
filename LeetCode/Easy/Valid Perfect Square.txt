Link: https://leetcode.com/problems/valid-perfect-square/
Language: C#



public class Solution {
public bool IsPerfectSquare(int num)
        {
            if (num == 0)
                return false;
            if (num == 1)
                return true;
            if (num > 1)
            {
                for (int i = 2; (long) i*i <= num; i++)
                {
                    if (i * i == num)
                        return true;
                }
            }         
            return false;
        }
}
